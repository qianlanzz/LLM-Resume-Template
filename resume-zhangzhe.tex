\sectionTitle{比赛经历}{\faTrophy}
\begin{itemize}
  \item \textbf{KDD Cup 2024 学术论文问答检索挑战赛} \hfill 2024.05 -- 2024.07
  \begin{itemize}
    \item \textbf{任务背景}: 面向学术问答场景，从海量论文中检索与专业问题最匹配的候选论文，核心挑战在于复杂 Query 冗长、语义匹配困难。
    \item \textbf{方案设计}: 设计``问题精炼 + 向量召回 + 重排序''两阶段检索方案——基于 Llama3 对 Query 精炼总结，使用 BGE M3 统一向量化，结合 FAISS 召回 Top100 候选，再引入 BGE-Reranker 精排筛选 Top20；通过 Hard Example Mining 构建微调数据，对 Reranker 进行专项优化，提升困难样本区分能力。
    \item \textbf{最终成果}: 官方测试集检索分数从 0.138 提升至 0.166，较基线提升约 20.3\%。
  \end{itemize}

  \item \textbf{``天翼云息壤杯'' AI 大赛：大语言模型数学推理能力提升} \hfill 2024.11 -- 2025.07
  \begin{itemize}
    \item \textbf{任务背景}: 基于 Qwen3-8B 开展微调训练，提升模型对复杂数学问题与深度推理的理解求解能力。
    \item \textbf{方案设计}: 针对数学推理长链路推导困难、监督样本质量不足的问题，设计``数据蒸馏 + 质量筛选 + 参数高效微调''整体方案——基于 MetaMathQA 构建数据集，利用 Qwen2.5-Math-72B 蒸馏生成 10 万条高质量 CoT 样本，结合 Math-Verify 与 Qwen2.5-72B 评审模型做答案校验。
    \item \textbf{工程落地}: 在 8 卡昇腾 910B 64G 环境完成 Qwen3-8B 的 LoRA 微调，结合 vLLM 优化推理性能，实现训推一体化部署。
    \item \textbf{最终成果}: 微调后客观数学题正确率从 38\% 提升至 62\%，提升幅度达 63\%。
  \end{itemize}
\end{itemize}

\sectionTitle{实习经历}{\faBriefcase}
\begin{itemize}
  \item \textbf{中国电子科技集团公司第二十八研究所（智能模型资源管理平台研发实习）} \hfill 2025.09 -- 至今
  \begin{itemize}

    \item \textbf{语义向量检索引擎}: 针对平台模型资源数量增长后关键词检索召回率不足的问题，基于 Qwen3-Embedding-0.6B 与 FAISS 搭建语义检索服务，支持用户以自然语言任务描述检索模型库；设计增量索引更新机制，通过 Django Signal 监听模型元数据及关联关系变更自动触发重建，避免全量重建开销；索引读写采用 clone-on-read 快照隔离保证搜索无锁并发，持久化通过原子文件替换保证崩溃安全；上线后模型检索从纯关键词匹配升级为语义级召回，用户定位目标模型的平均交互轮次显著减少。

  \end{itemize}

\sectionTitle{项目经历}{\faCode}
\begin{itemize}
  \item \textbf{OCR 运维诊断 Agent（DeerFlow / LangGraph）} \hfill 2026.01 -- 至今
  \begin{itemize}
    \item \textbf{项目背景}: 面向 OCR 服务运维场景，运维人员需要快速定位告警根因并获取处置建议，传统人工排查效率低、依赖经验。
    \item \textbf{诊断链路设计}: 基于 DeerFlow / LangGraph 构建 Triage $\rightarrow$ Diagnosis $\rightarrow$ Report 三阶段工作流，支持自然语言输入告警或排障问题，自动完成告警分类、证据收集、根因分析与处置建议生成。
    \item \textbf{工具接入与分析}: 接入 Prometheus 指标查询、Docker 容器状态与日志采集、历史案例检索等诊断工具，覆盖 API P95、队列 waiting/failed/workers、CPU/内存占用等关键证据；Diagnosis 阶段采用规则诊断 + LLM 校对策略，降低纯模型推断的误判风险。
    \item \textbf{可观测性}: 开发 AIOps 诊断台，基于 SSE 展示 Agent 各阶段输入输出与分析过程，支持诊断链路回放，便于故障复盘与演示。
  \end{itemize}

  \item \textbf{教辅录题审核一体化系统} \hfill 2025.06 -- 至今
  \begin{itemize}
    \item \textbf{项目背景}: 原有教辅录题依赖 Excel + 多工具协作，流程碎片化、效率低，需要一套打通上传、录题、质检、入库全链路的一体化系统。
    \item \textbf{主流程建设}: 基于 Vue 3 + TypeScript、Fastify + Prisma + PostgreSQL + Redis 开发录题工作台及后端服务，支持 PDF 双栏浏览、框选 OCR、公式渲染、草稿自动保存，打通教材上传到结果入库的完整链路。
    \item \textbf{结构化自动切题}: 主导自动切题模块，基于 MinerU 的块级坐标解析与目录分段能力将教材 PDF 转换为结构化内容，结合大模型完成题号、题干、选项、答案、解析抽取，基于``章节标题 + 题号''实现题答配对，生成包含编号、页码、坐标、题型等字段的标注结果回写工作台。
    \item \textbf{稳定性治理}: 基于 BullMQ 承接 PDF 拆页、MinerU 解析与自动切题等长任务，避免重任务阻塞工作台；补充队列监控告警、草稿缓存与历史恢复、提交前校验等机制，降低卡顿、丢稿和返工风险。
  \end{itemize}
\end{itemize}
